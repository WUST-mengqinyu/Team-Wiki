    A. B-Suffix ArrayLink

    solved by: Verly 03:25:17(-8) upsolved by:

    题意：给一个全为ab的字符串，对于其中任一后缀，第i位的值为i到前一个相同字符的距离（如果s[i]第一次出现，值为0），对后缀赋的值排序。

    对于某一个后缀串s‘，可以将其分为a[*a]b|xxxx或b[*b]a|xxxx,对于长度为len的第一段，其值为0[(len-2)*1]0 + sa[pos+len]，对于每种长度的len，对sa[pos+len]排序，按len从小到大输出即可。

    Code
    #include <bits/stdc++.h>

using namespace std;
typedef long long ll;

const int maxn = 3e5 + 10;
int s[maxn];
char b[maxn];
int sa[maxn], t[maxn], t2[maxn], c[maxn], n;

//build_sa(n + 1, 130), sa, height下标从1开始,rk下标从0开始
void build_sa(int n, int m) {
int *x = t, *y = t2;
for (int i = 0; i < m; i++) c[i] = 0;
for (int i = 0; i < n; i++) c[x[i] = s[i]]++;
for (int i = 1; i < m; i++) c[i] += c[i - 1];
for (int i = n - 1; i >= 0; i--) sa[--c[x[i]]] = i;
for (int k = 1; k <= n; k <<= 1) {
    int p = 0;
    for (int i = n - k; i < n; i++) y[p++] = i;
    for (int i = 0; i < n; i++) if (sa[i] >= k) y[p++] = sa[i] - k;
    for (int i = 0; i < m; i++) c[i] = 0;
    for (int i = 0; i < n; i++) c[x[y[i]]]++;
    for (int i = 0; i < m; i++) c[i] += c[i - 1];
    for (int i = n - 1; i >= 0; i--) sa[--c[x[y[i]]]] = y[i];
    swap(x, y);
    p = 1;
    x[sa[0]] = 0;
    for (int i = 1; i < n; i++)
        x[sa[i]] = y[sa[i - 1]] == y[sa[i]] && y[sa[i - 1] + k] == y[sa[i] + k] ? p - 1 : p++;
    if (p >= n) break;
    m = p;
}
}

int rk[maxn];

void getHeight() {
for(int i = 1; i <= n; i++) rk[sa[i]] = i;
}

int nex[maxn];
vector<pair<int, int>> resp[maxn];
int resq[maxn];

int main() {
    while (scanf("%d", &n) != EOF) {
        scanf("%s", b);
        for (int i = 0; i <= n; i++) resp[i].clear(), resq[i] = -1, nex[i] = 0;
        s[n] = s[n + 1] = 0;
        int lasa = -1, lasb = -1;
        for (int i = 0; i < n; i++) {
            if (b[i] == 'a') {
                if (lasa == -1) s[i] = 50;
                else s[i] = i - lasa + 50;
                lasa = i;
            } else if (b[i] == 'b') {
                if (lasb == -1) s[i] = 50;
                else s[i] = i - lasb + 50;
                lasb = i;
            }
        }
        lasa = -1, lasb = -1;
        for (int i = n - 1; i >= 0; i--) {
            if (b[i] == 'a') {
                nex[i] = lasb;
                lasa = i;
            } else if (b[i] == 'b') {
                nex[i] = lasa;
                lasb = i;
            }
        }
        build_sa(n + 1, n + 55);
        for (int i = 0; i <= n + 1; ++i) rk[i] = 0;
        getHeight();
        for (int i = 0; i < n - 1; i++) {
            if (nex[i] == -1) resq[n - i] = i;
            else {
                int len = nex[i] - i + 1;
                resp[len].push_back({rk[nex[i] + 1], i});
            }
        }
        for (int i = 0; i <= n; i++) sort(resp[i].begin(), resp[i].end());
        printf("%d", n);
        for (int i = 0; i <= n; i++) {
            for (auto x : resp[i]) printf(" %d", x.second + 1);
            if (resq[i] != -1) printf(" %d", resq[i] + 1);
        }
        puts("");
    }
    return 0;
}

    H. Minimum-cost FlowLink

    upsolved by: Verly

    题意：给一张网络流图，q次询问，每次给每一条边u/v的流量，问1到n流1个单位流量的最小费用。

    对原图跑1到n的网络流，考虑贪心，每次bellman-ford代表当前残量网络从1到n流1个单位所需的最小花费，对每个询问的所需流量取前缀即可。

    Code
    #include <bits/stdc++.h>

using namespace std;
typedef long long ll;

const int maxn = 10000 + 10;
const int inf = 0x3f3f3f3f;

struct Edge { int from, to, cap, flow, cost; };

struct MCMF
{
    int n, m;
    vector<Edge> edges;
    vector<int> G[maxn];
    bool inq[maxn];
    int dis[maxn], path[maxn], a[maxn];
    vector<int> ve;

    void init(int n)
    {
        this->n = n;
        for(int i = 0;i <= n;i ++)
            G[i].clear();
        edges.clear();
        ve.clear();
    }

    void addEdge(int from, int to, int cap, int cost)
    {
        edges.push_back(Edge{from, to, cap, 0, cost});
        edges.push_back(Edge{to, from, 0, 0, -cost});
        m = edges.size();
        G[from].push_back(m - 2);
        G[to].push_back(m - 1);
    }

    bool Bellman_Ford(int s, int t, int& flow, int& cost)
    {
        for(int i = 0; i<= n; i++) dis[i] = inf;
        memset(inq, 0, sizeof inq);
        dis[s]=0, inq[s]=true, path[s]=0, a[s]=inf;
        queue<int> Q;
        Q.push(s);
        while(!Q.empty())
        {
            int u = Q.front(); Q.pop();
            inq[u] = false;
            for(int i = 0; i < G[u].size(); i++)
            {
                Edge& e = edges[G[u][i]];
                if(e.cap > e.flow && dis[e.to] > dis[u] + e.cost)
                {
                    dis[e.to] = dis[u] + e.cost;
                    path[e.to] = G[u][i];
                    a[e.to] = min(a[u], e.cap - e.flow);
                    if(!inq[e.to])
                    {
                        Q.push(e.to);
                        inq[e.to] = true;
                    }
                }
            }
        }
        if(dis[t] == inf) return false;     //求最小费用最大流
        //if(1ll * dis[t] * a[t] > 0) return false; 求可行流最小费用，因此当费用增量大于0时不继续增加流量
        flow += a[t];
        cost += dis[t] * a[t];
        for(int u = t; u != s; u = edges[path[u]].from)
        {
            edges[path[u]].flow += a[t];
            edges[path[u] ^ 1].flow -= a[t];
        }
        return true;
    }

    void mincostMaxFlow(int s, int t)
    {
        int flow = 0, cost = 0;
        while(Bellman_Ford(s, t, flow, cost))
        {
            ve.push_back(cost);
            cost = 0;
        }
    }
}ans;

int main()
{
    int n, m;
    while(scanf("%d%d", &n, &m) != EOF)
    {

        int S = 1, T = n;
        ans.init(T);
        for(int i = 0; i < m; i ++)
        {
            int u, v, w;
            scanf("%d%d%d", &u, &v, &w);
            ans.addEdge(u, v, 1, w);
        }
        ans.mincostMaxFlow(S, T);
        int q;
        scanf("%d", &q);
        while(q --)
        {
            int u, v;
            scanf("%d%d", &u, &v);
            ll resx = 0, resy = v;
            int tot = 0;
            for(auto x : ans.ve)
            {
                if(tot == v) break;
                int now = min(v - tot, u);
                resx += 1ll * x * now;
                tot += now;
            }
            if(tot < v) puts("NaN");
            else
            {
                long long g = __gcd(resx, resy);
                printf("%lld/%lld\n", resx/g, resy/g);
            }
        }
    }
    return 0;
}

    I. 1 or 2Link

    upsolved by: Verly

    题意：给一张n个点m条边的图(1≤n≤50, 1≤m≤100)，问是否存在生成子图，满足点i的度数为de[i](1≤d_i≤2)

    对于原图的点拆点表示度数，对连接u，v的边拆点e,e'，连边u->e, v->e', e->e'跑一般图匹配，考虑以下情况: - u,v都不与其它边匹配，此时选中u->e, v->e'，maxmatch为2，代表边e被选中，并为u,v各提供了一个度数 - u,v有至少一个点与其他边匹配，maxmatch为1

    可以看到当某条边被选中时，对最大匹配的贡献+1

    sum=\sum_{i=1}^n de_i为奇数显然无解，否则当maxmatch=sum/2+m时存在可行解。
