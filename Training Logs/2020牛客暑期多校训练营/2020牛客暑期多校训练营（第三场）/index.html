    D. Points Construction ProblemLink

    upsolved by: Verly

    题意:在无限大的格子中,原本所有格子都是白色的,把n个格子涂黑,满足上下左右相邻的两个点对为异色的个数为m个

    对于m,容易得到$m≤4n$且m只能为偶数,找到一组ab,使$n≤ab$且$abs(a-b)≤1$,$2*(a-b)$即为m的下界

    首先填充长方形的左边缘和下边缘,此时异色点对数为$2*(a-b)$,之后在往长方形内放点,异色点对数不变

    当异色点对数小于m时,不断将非左边缘和下边缘的点单独丢到远处,此时异色点对数+4,如果异色点对数>m,一定只比m大2,把远处一个单独的点放到0,0下方连接即可

    如果非左边缘和下边缘的点全部丢完时异色点对数仍然小于m,将左边缘最上面或者下边缘最右侧的点丢向远处,每次异色点对+2,直到异色点对数=$4*n$

    F. Fraction Construction ProblemLink

    solved by: ffacs 04:47:35(-4)

    题意：给定一个分数$\frac a b$，求c,d,e,f满足$\frac c e - \frac d f = \frac a b(d, f<b,c,e≤ 4×10^{12})$

    如果ab不互质，gcd(a,b)+a, b, gcd(a, b), b即为答案

    对于互质的情况，如果b的不同质因数个数只有1个，即$b=p^k$，由于a,b互质,$\frac c d - \frac e f$化简后必须是$\frac a b$,但d,f<b,df的质因数分解质因数指数都小于k,得到矛盾,所以无解

    如果b的不同质因数个数大于1,找到df满足$df=b$且$__gcd(d,f)=1$,有$\frac {cf-ed} {df}=\frac a b$,用exgcd求$cf-ed=a$

    G. Operating on a GraphLink

    upsolved by: Verly

    题意:给一张n个点m条边的图,初始时点i的颜色为i，进行q次操作，每次将颜色为i的点周围的点染成i，求q次操作后n个点的颜色。

    考虑每个点所连出的边最多只会被访问一次，所以只要维护某个颜色当前还没有给相邻点染色的点，用链表维护边集，用并查集维护点的颜色，每次染色时合并链表即可，复杂度$O(n)$

    ??? note “Code” ```cpp #include 

    using namespace std;
typedef long long ll;

const int maxn = 8e5 + 10;
list<int> edge[maxn];
int pre[maxn];

int Find(int x) { return x == pre[x]? x : pre[x] = Find(pre[x]); }

int main()
{
    int t;
    scanf("%d", &t);
    while(t --)
    {
        int n, m;
        scanf("%d%d", &n, &m);
        for(int i = 0; i < n; i ++)
        {
            pre[i] = i;
            edge[i].clear();
        }
        while(m --)
        {
            int u, v;
            scanf("%d%d", &u, &v);
            edge[u].push_back(v);
            edge[v].push_back(u);
        }
        int q;
        scanf("%d", &q);
        for(int i = 0; i < q; i ++)
        {
            int x;
            scanf("%d", &x);
            if(x != Find(x)) continue;
            list<int> tmp;
            for(auto v : edge[x])
            {
                int fv = Find(v);
                if(fv == x) continue;
                pre[fv] = x;
                tmp.splice(tmp.end(), edge[fv]);
            }
            swap(edge[x], tmp);
        }
        for(int i = 0; i < n; i ++)
            printf("%d%c", Find(i), " \n"[i == n - 1]);
    }
    return 0;
}

    H. Sort the Strings RevisionLink

    upsolved by: Verly

    题意：长度为n的串$s_0$为01234567890123456789...，给定一个0～n-1的排列p和序列d，第i次操作表示把$s_{i-1}$的第$p_i$位修改成$d_i$并生成$s_i$，对$s_0$到$s_n$按字典序排序，$r_i$表示$s_i$在排序后的位置（从0开始），求$\sum_{i=0}^n (r_i*10000019^i)\ mod \ 1000000007$

    考虑以$p_i$在区间内的最小值进行分治，对于$p_i$最小值为pos的区间l,r，如果p[pos]%10!=d[pos]，有

    • p[pos]%10<d[pos]，rk[l~pos-1]+=r-pos
    • p[pos]%10>d[pos],rk[pos~r]+=pos-l+1

    对于每个子区间递归做下去即可。

    ??? note “Code” ```cpp #include 

    using namespace std;
typedef long long ll;

const int maxn = 2e6 + 10;
const int mod = 1e9 + 7;

int n;
int p[maxn], d[maxn], ls[maxn], rs[maxn], st[maxn], rk[maxn];
ll pw[maxn] = {1};

void rand_p()
{
    int seed, a, b, mod;
    scanf("%d%d%d%d", &seed, &a, &b, &mod);
    for(int i = 0; i < n; i ++) p[i] = i;
    for(int i = 1; i < n; i ++)
    {
        swap(p[seed % (i + 1)], p[i]);
        seed = (1ll * seed * a + b) % mod;
    }
}

void rand_d()
{
    int seed, a, b, mod;
    scanf("%d%d%d%d", &seed, &a, &b, &mod);
    for(int i = 0; i < n; i ++)
    {
        d[i] = seed % 10;
        seed = (1ll * seed * a + b) % mod;
    }
}

void dfs(int l, int r, int root, int val)
{
    if(p[root] == n || l >= r) 
    {
        for(int i = l; i <= r; i ++) rk[i] = val + i - l;
        return;
    }
    dfs(l, root, ls[root], (p[root] % 10 > d[root]) ? (val + r - root) : val), dfs(root + 1, r, rs[root], (p[root] % 10 < d[root]) ? (val + root - l + 1) : val);
}

int main()
{
    int t;
    scanf("%d", &t);
    for(int i = 1; i < maxn; i ++) pw[i] = (pw[i - 1] * 10000019) % mod;
    while(t --)
    {
        scanf("%d", &n);
        rand_p(); rand_d();
        for(int i = 0; i < n; i ++) if(p[i] % 10 == d[i]) p[i] = n;
        int top = 0;
        for(int i = 0; i < n; i ++)
        {
            int k = top;
            while(k > 0 && p[st[k]] > p[i]) k --;
            if(k) rs[st[k]] = i;
            if(k < top) ls[i] = st[k + 1];
            st[++ k] = i;
            top = k;
        }
        dfs(0, n, st[1], 0);
        ll ans = 0;
        for(int i = 0; i <= n; i ++) ans = (ans + rk[i] * pw[i]) % mod; 
        printf("%lld\n", ans);
    }
    return 0;
}

    string s;

    int flag = 0;
    for (int i = 0; i < s.length(); ++i) {
        if (s[i] >= 'A' && s[i] <= 'Z') {
            s[i] += 'a' - 'A';
        }
        if (s[i] == t[i]) {
            flag ++;
        }
        if (flag == t.length()) break;
    }
    W(flag == t.length() ? "lovely" : "ugly");
    return 0;
}
